TileEngine engine = new TileEngine();
            MapUI mapui = new MapUI();
            Shop shop = new Shop();
            LevelManager manager = new LevelManager();
            ConsoleKeyInfo keyInput = new ConsoleKeyInfo();
            manager.SelectLevel(1);
            int levelCheck = manager.currentLevel;
            int currentPlayerGold = 50;
            int currentPlayerHP = 20;
            int maxPlayerHP = 50;
            engine.SpawnPlayer(manager.playerStartPosX, manager.playerStartPosY);
            mapui.UILevelUpdate(manager.levelData, manager.mapHeight, manager.mapWidth, engine.currentPlayerPosX, engine.currentPlayerPosY);
            mapui.DrawShopButton();
            while (true)
            {
                keyInput = Console.ReadKey();
                engine.PlayerMovement(keyInput, manager.levelData, manager.mapWidth);
                levelCheck = engine.TileEvents(manager.levelData,engine.currentPlayerPosY,engine.currentPlayerPosX,manager.mapWidth,manager.currentLevel);
                if(manager.currentLevel != levelCheck)
                {
                    manager.currentLevel = levelCheck;
                    manager.SelectLevel(manager.currentLevel);
                    engine.SpawnPlayer(manager.playerStartPosX, manager.playerStartPosY);
                }
                if(keyInput.Key == ConsoleKey.S)
                {
                    shop.StartShop(currentPlayerGold,currentPlayerHP,maxPlayerHP);
                }
                mapui.UILevelUpdate(manager.levelData, manager.mapHeight, manager.mapWidth, engine.currentPlayerPosX, engine.currentPlayerPosY);
                mapui.DrawShopButton();
            }
